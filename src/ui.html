<% if (htmlWebpackPlugin.options.mode === 'development') { %> <% } %> /** * Color Token Enhancer - Can be added to the existing UI * * This is a standalone script that enhances color token visualization * It can be included in the existing UI without changing the core functionality */ // Minimal reference to avoid conflicts const ColorTokenEnhancer = { // Main function to enhance color token displays enhance: function() { // Find all elements that look like color tokens this.findAndEnhanceColorTokens(); // Set up a mutation observer to handle dynamically added content this.setupObserver(); console.log('Color token enhancer initialized'); }, // Find and enhance color tokens in the current DOM findAndEnhanceColorTokens: function() { // Find potential color values in the UI const colorElements = document.querySelectorAll('pre, .token-value'); colorElements.forEach(element => { const text = element.textContent || ''; // Look for color values - hex, rgb, hsl const hexMatches = text.match(/#[0-9a-fA-F]{3,8}\b/g); const rgbMatches = text.match(/rgb\(.*?\)/g) || text.match(/rgba\(.*?\)/g); const hslMatches = text.match(/hsl\(.*?\)/g) || text.match(/hsla\(.*?\)/g); // Find reference values like {colors.primary} const refMatches = text.match(/\{[^}]+\}/g); if (hexMatches || rgbMatches || hslMatches) { // It's a color value! this.enhanceColorElement(element, hexMatches || rgbMatches || hslMatches); } else if (refMatches) { // It's a reference! this.enhanceReferenceElement(element, refMatches[0]); } }); }, // Enhance a color element with a color chip enhanceColorElement: function(element, colorValues) { if (!colorValues || !colorValues.length) return; // Get the first color value const colorValue = colorValues[0]; // Don't modify if it's already been enhanced if (element.querySelector('.color-value-chip')) return; // Create a color chip const colorChip = document.createElement('span'); colorChip.className = 'color-value-chip'; colorChip.style.backgroundColor = colorValue; // Add it before the text element.insertBefore(colorChip, element.firstChild); }, // Enhance a reference element enhanceReferenceElement: function(element, refValue) { // Don't modify if it's already been enhanced if (element.querySelector('.reference-label')) return; // Add a reference label const refLabel = document.createElement('span'); refLabel.className = 'reference-label'; refLabel.textContent = 'Ref'; // Add it after the text element.appendChild(refLabel); // Try to get the parent container (to add the split swatch) const container = this.findParentContainer(element); if (container) { this.addSplitSwatch(container, refValue); } }, // Find a suitable parent container findParentContainer: function(element) { // Try to find a container element that we can add a swatch to let current = element.parentElement; while (current && !current.classList.contains('token-preview-item') && !current.classList.contains('color-preview-item')) { current = current.parentElement; } return current; }, // Add a split swatch showing both reference and value addSplitSwatch: function(container, refValue) { // Don't add if there's already a swatch if (container.querySelector('.split-swatch')) return; // Try to find the resolved value in the DOM const resolvedValueEl = container.querySelector('.resolved-value'); const resolvedColor = resolvedValueEl ? resolvedValueEl.textContent.trim() : '#cccccc'; // Create the split swatch const splitSwatch = document.createElement('div'); splitSwatch.className = 'split-swatch'; const top = document.createElement('div'); top.className = 'top'; const bottom = document.createElement('div'); bottom.className = 'bottom'; bottom.style.backgroundColor = resolvedColor; const refIcon = document.createElement('div'); refIcon.className = 'reference-icon'; refIcon.textContent = '↗'; splitSwatch.appendChild(top); splitSwatch.appendChild(bottom); splitSwatch.appendChild(refIcon); // Add it to the container container.insertBefore(splitSwatch, container.firstChild); }, // Setup mutation observer to enhance dynamically added elements setupObserver: function() { const observer = new MutationObserver(mutations => { let shouldEnhance = false; mutations.forEach(mutation => { if (mutation.type === 'childList' && mutation.addedNodes.length) { shouldEnhance = true; } }); if (shouldEnhance) { this.findAndEnhanceColorTokens(); } }); observer.observe(document.body, { childList: true, subtree: true }); } }; // Initialize when the DOM is ready document.addEventListener('DOMContentLoaded', function() { // Wait a bit to make sure the UI is fully loaded setTimeout(() => { ColorTokenEnhancer.enhance(); }, 1000); }); // Also run it now in case the DOM is already loaded if (document.readyState === 'complete' || document.readyState === 'interactive') { setTimeout(() => { ColorTokenEnhancer.enhance(); }, 1000); }
